14class BayesianCalibration(TorchDeviceMixin): 

15 """ 

16 Bayesian calibration estimates the probability distribution over input parameters 

17 given observed data, providing uncertainty estimates. 

18 """ 

19 

20 def __init__( # noqa: PLR0913 

21 self, 

22 emulator: Emulator, 

23 parameter_range: dict[str, tuple[float, float]], 

24 observations: dict[str, TensorLike], 

25 observation_noise: float | dict[str, float] = 0.1, 

26 calibration_params: list[str] | None = None, 

27 device: DeviceLike | None = None, 

28 log_level: str = "progress_bar", 

29 ): 

30 """ 

31 Initialize the HMC calibration object. 

32 

33 Parameters 

34 ---------- 

35 emulator: Emulator 

36 Fitted Emulator object. 

37 parameters_range : dict[str, tuple[float, float]] 

38 A dictionary mapping input parameter names to their (min, max) ranges. 

39 observations: dict[str, TensorLike] 

40 A dictionary of observations for each output. 

41 observation_noise: float | dict[str, float] 

42 A single value or a dictionary of values (one per output). Defaults to 0.1. 

43 calibration_params: list[str] | None 

44 Optional list of input parameters to calibrate. Any parameters that are not 

45 listed will be set to the midpoint value of their parameter range. If None, 

46 will calibrate all input parameters. Defaults to None. 

47 device: DeviceLike | None 

48 The device to use. If None, the default torch device is returned. 

49 TODO: do we need to do anything more to ensure the device is correctly 

50 handled for the pyro model? 

51 log_level: str 

52 Logging level for the calibration. Can be one of: 

53 - "progress_bar": shows a progress bar during batch simulations 

54 - "debug": shows debug messages 

55 - "info": shows informational messages 

56 - "warning": shows warning messages 

57 - "error": shows error messages 

58 - "critical": shows critical messages 

59 

60 Notes 

61 ----- 

62 The model assumes: 

63 - Uniform priors for calibrated parameters (bounds given by `parameters_range`) 

64 - Gaussian likelihood with no correlation between outputs 

65 All non-calibrated parameters are set to a constant value. This is chosen as the 

66 midpoint value of `parameters_range`. 

67 """ 

68 TorchDeviceMixin.__init__(self, device=device) 

69 self.parameter_range = parameter_range 

70 if calibration_params is None: 

71 calibration_params = list(parameter_range.keys()) 

72 self.calibration_params = calibration_params 

73 self.emulator = emulator 

74 self.emulator.device = self.device 

75 self.output_names = list(observations.keys()) 

76 self.logger, self.progress_bar = get_configured_logger(log_level) 

77 self.logger.info( 

78 "Initializing BayesianCalibration with parameters: %s", 

79 self.calibration_params, 

80 ) 

81 

82 # Check observation tensors are 1D (convert if 0D) 

83 processed_observations = {} 

84 obs_lengths = [] 

85 for output, obs in observations.items(): 

86 if obs.ndim == 0: 

87 corrected_obs = obs.unsqueeze(0) 

88 self.logger.debug( 

89 "Observation for output '%s' converted from 0D to 1D.", 

90 output, 

91 ) 

92 elif obs.ndim > 1: 

93 self.logger.error("Tensor for output '%s' is not 1D.", output) 

94 raise ValueError(f"Tensor for output '{output}' is not 1D.") 

95 else: 

96 corrected_obs = obs 

97 processed_observations[output] = corrected_obs.to(self.device) 

98 obs_lengths.append(corrected_obs.shape[0]) 

99 if len(set(obs_lengths)) != 1: 

100 msg = "All outputs must have the same number of observations." 

101 self.logger.error(msg) 

102 raise ValueError(msg) 

103 self.observations = processed_observations 

104 self.n_observations = obs_lengths[0] 

105 self.logger.info("Processed observations for outputs: %s", self.output_names) 

106 

107 # Save observation noise as {output: value} dictionary 

108 if isinstance(observation_noise, float): 

109 self.observation_noise = dict.fromkeys( 

110 self.output_names, 

111 torch.tensor(observation_noise).to(self.device), 

112 ) 

113 self.logger.debug("Observation noise set as float: %s", observation_noise) 

114 elif isinstance(observation_noise, dict): 

115 self.observation_noise = observation_noise 

116 self.logger.debug("Observation noise set as dict: %s", observation_noise) 

117 else: 

118 msg = "Noise must be either a float or a dictionary of floats." 

119 self.logger.error(msg) 

120 raise ValueError(msg)

122 def _get_kernel(self, sampler: str, **sampler_kwargs): 

123 """ 

124 Get the appropriate MCMC kernel based on sampler choice. 

125 """ 

126 sampler = sampler.lower() 

127 

128 if sampler == "nuts": 

129 self.logger.debug("Using NUTS kernel.") 

130 return NUTS(self.model, **sampler_kwargs) 

131 if sampler == "hmc": 

132 step_size = sampler_kwargs.pop("step_size", 0.01) 

133 trajectory_length = sampler_kwargs.pop("trajectory_length", 1.0) 

134 self.logger.debug( 

135 "Using HMC kernel with step_size=%s, trajectory_length=%s", 

136 step_size, 

137 trajectory_length, 

138 ) 

139 return HMC( 

140 self.model, 

141 step_size=step_size, 

142 trajectory_length=trajectory_length, 

143 **sampler_kwargs, 

144 ) 

145 if sampler == "metropolis": 

146 self.logger.debug("Using Metropolis (RandomWalkKernel).") 

147 return RandomWalkKernel(self.model, **sampler_kwargs) 

148 self.logger.error("Unknown sampler: %s", sampler) 

149 raise ValueError(f"Unknown sampler: {sampler}")

151 def model(self, predict: bool = False): 

152 """ 

153 Pyro model. 

154 

155 Parameters 

156 ---------- 

157 predict: bool 

158 Whether to run the model with existing samples to generate posterior 

159 predictive distribution. Used with `pyro.infer.Predictive`. 

160 """ 

161 

162 # Pre-allocate tensor for all input parameters, shape [1, n_inputs] 

163 param_list = [] 

164 # Each param is either sampled (if calibrated) or set to a constant value 

165 for param in self.parameter_range: 

166 if param in self.calibration_params: 

167 # Sample from uniform prior 

168 min_val, max_val = self.parameter_range[param] 

169 sampled_val = pyro.sample(param, dist.Uniform(min_val, max_val)) 

170 param_list.append(sampled_val.to(self.device)) 

171 else: 

172 # Set to midpoint value in parameter range 

173 min_val, max_val = self.parameter_range[param] 

174 midpoint_val = (min_val + max_val) / 2 

175 param_list.append(torch.tensor(midpoint_val, device=self.device)) 

176 full_params = torch.stack(param_list, dim=0).unsqueeze(0).float() 

177 

178 # Get emulator prediction 

179 output = self.emulator.predict(full_params, with_grad=True) 

180 if isinstance(output, TensorLike): 

181 pred_mean = output.to(self.device) 

182 elif isinstance(output, DistributionLike): 

183 pred_mean = output.mean.to(self.device) 

184 else: 

185 msg = "The emulator did not return a tensor or a distribution object." 

186 self.logger.error(msg) 

187 raise ValueError(msg) 

188 

189 # Likelihood 

190 for i, output in enumerate(self.output_names): 

191 if not predict: 

192 with pyro.plate(f"data_{output}", self.n_observations): 

193 pyro.sample( 

194 output, 

195 dist.Normal(pred_mean[0, i], self.observation_noise[output]), 

196 obs=self.observations[output], 

197 ) 

198 else: 

199 with pyro.plate(f"data_{output}", self.n_observations): 

200 pyro.sample( 

201 output, 

202 dist.Normal(pred_mean[0, i], self.observation_noise[output]), 

203 )

205 def run_mcmc( 

206 self, 

207 warmup_steps: int = 500, 

208 num_samples: int = 1000, 

209 num_chains: int = 1, 

210 initial_params: dict[str, TensorLike] | None = None, 

211 sampler: str = "nuts", 

212 **sampler_kwargs, 

213 ) -> MCMC: 

214 """ 

215 Run Markov Chain Monte Carlo (MCMC). Defaults to using the NUTS sampler. 

216 

217 Parameters 

218 ---------- 

219 warmup_steps: int 

220 Number of warm up steps to run per chain (i.e., burn-in). These samples are 

221 discarded. Defaults to 500. 

222 num_samples: int 

223 Number of samples to draw after warm up. Defaults to 1000. 

224 num_chains: int 

225 Number of parallel chains to run. Defaults to 1. 

226 initial_params: dict[str, TensorLike] | None 

227 Optional dictionary specifiying initial values for each calibration 

228 parameter per chain. The tensors must be of length `num_chains`. 

229 sampler: str 

230 The MCMC kernel to use, one of "hmc", "nuts" or "metropolis". 

231 

232 Returns 

233 ------- 

234 MCMC 

235 The Pyro MCMC object. Methods include `summary()` and `get_samples()`. 

236 """ 

237 

238 # Check initial param values match number of chains 

239 

240 if initial_params is not None: 

241 for param, init_vals in initial_params.items(): 

242 if init_vals.shape[0] != num_chains: 

243 msg = ( 

244 "An initial value must be provided for each chain, parameter " 

245 f"{param} tensor only has {init_vals.shape[0]} values." 

246 ) 

247 self.logger.error(msg) 

248 raise ValueError(msg) 

249 self.logger.debug( 

250 "Initial parameters provided for MCMC: %s", initial_params 

251 ) 

252 

253 # Run NUTS 

254 kernel = self._get_kernel(sampler, **sampler_kwargs) 

255 mcmc = MCMC( 

256 kernel, 

257 warmup_steps=warmup_steps, 

258 num_samples=num_samples, 

259 num_chains=num_chains, 

260 # If None, init values are sampled from the prior. 

261 initial_params=initial_params, 

262 # Multiprocessing 

263 mp_context="spawn" if num_chains > 1 else None, 

264 ) 

265 self.logger.info("Starting MCMC run.") 

266 mcmc.run() 

267 self.logger.info("MCMC run completed.") 

268 return mcmc

292 ) -> az.InferenceData: 

293 """ 

294 Convert MCMC object to Arviz InferenceData object for plotting. 

295 

296 Parameters 

297 ---------- 

298 mcmc: MCMC 

299 The MCMC object. 

300 posterior_predictive: bool 

301 Whether to include posterior predictive samples. Defaults to False. 

302 

303 Returns 

304 ------- 

305 az.InferenceData 

306 """ 

307 pp_samples = None 

308 if posterior_predictive: 

309 self.logger.info("Including posterior predictive samples in Arviz output.") 

310 pp_samples = self.posterior_predictive(mcmc) 

311 

312 # Need to create dataset manually for Metropolis Hastings 

313 # This is because az.from_pyro expects kernel with `divergences` 

314 if isinstance(mcmc.kernel, RandomWalkKernel): 

315 self.logger.debug(

317 ) 

318 if posterior_predictive: 

319 az_data = az.InferenceData( 

320 posterior=az.convert_to_dataset( 

321 mcmc.get_samples(group_by_chain=True) 

322 ), 

323 posterior_predictive=az.convert_to_dataset(pp_samples), 

324 observed_data=az.convert_to_dataset(self.observations), 

325 ) 

326 else: 

327 az_data = az.InferenceData( 

328 posterior=az.convert_to_dataset( 

329 mcmc.get_samples(group_by_chain=True) 

330 ), 

331 ) 

332 else: 

333 self.logger.debug("Using az.from_pyro for conversion.") 

334 az_data = az.from_pyro(mcmc, posterior_predictive=pp_samples) 

335 

336 self.logger.info("Arviz InferenceData conversion complete.") 

337 return az_data
