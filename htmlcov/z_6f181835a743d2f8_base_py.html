46 def _expanded_basis_matrix(self, y: TensorLike) -> TensorLike: 

47 """Expanded basis matrix for the transformation of the number of samples in 

48 a given codomain `y` sample tensor. 

49 

50 Given `n` samples in `y`, this returns a Kronecker product of the identity 

51 matrix with the basis matrix, effectively expanding the basis matrix to match 

52 the number of samples in `y`. 

53 

54 The default implementation assumes that the transform has implemented the 

55 `_basis_matrix` property for a single sample. However, this can be overridden 

56 in subclasses to enable alternative approaches to generating the expanded basis 

57 matrix for a given set of samples. 

58 

59 Parameters 

60 ---------- 

61 y : TensorLike 

62 Input tensor of shape `(n, )` from which to compute the expanded 

63 basis matrix. The number of samples `n` is inferred from the shape of `y`. 

64 

65 Returns 

66 ------- 

67 TensorLike 

68 Expanded basis matrix. 

69 

70 Raises 

71 ------ 

72 RuntimeError 

73 If the transform has not been fitted yet. 

74 

75 """ 

76 self._check_is_fitted() 

77 return torch.kron(torch.eye(y.shape[0]), self._basis_matrix)

79 def _inverse_sample( 

80 self, y: GaussianLike, n_samples: int = 1000, full_covariance: bool = True 

81 ) -> GaussianLike: 

82 """Transforms a `GaussianLike` in the codomain to a `GaussianLike` in the domain 

83 through generating samples from `y` in the codomain and mapping those back 

84 to the original space `x`. 

85 

86 The empirical mean and covariance of the samples are computed, and a 

87 `GaussianLike` object in the domain is returned with these statistics. 

88 

89 Parameters 

90 ---------- 

91 y : GaussianLike 

92 The distribution in the codomain. 

93 n_samples : int, default=1000 

94 Number of samples to generate from the distribution `y`. 

95 full_covariance : bool, default=True 

96 If True, calculates a full covariance matrix from samples; otherwise, 

97 calculates only the diagonal of the covariance matrix. This is useful 

98 for a high-dimensional domain where full covariance might be 

99 computationally expensive. 

100 

101 Returns 

102 ------- 

103 GaussianLike 

104 A `GaussianLike` object representing the distribution in the domain with 

105 mean and covariance derived from the samples. 

106 

107 Raises 

108 ------ 

109 RuntimeError 

110 If covariance matrix cannot be made positive definite. 

111 

112 """ 

113 samples = self.inv(torch.stack([y.sample() for _ in range(n_samples)], dim=0)) 

114 assert isinstance(samples, TensorLike) 

115 mean = samples.mean(dim=0) 

116 cov = ( 

117 make_positive_definite(samples.view(n_samples, -1).T.cov()) 

118 if full_covariance 

119 else DiagLinearOperator(samples.view(n_samples, -1).var(dim=0)) 

120 ) 

121 return GaussianLike(mean, cov)

123 def _inverse_gaussian(self, y: GaussianLike) -> GaussianLike: 

124 r"""Transforms a `GaussianLike` in the codomain to a `GaussianLike` in the 

125 domain by applying the inverse of the transform to the mean and covariance of 

126 `y` in the codomain. 

127 

128 The default implementation computes the mean and covariance of `y` and: 

129 - applies the inverse transformation to the mean 

130 - transforms the covariance matrix using the expanded basis matrix 

131 

132 Parameters 

133 ---------- 

134 y : GaussianLike 

135 The distribution in the codomain to be transformed back to the domain. 

136 

137 Returns 

138 ------- 

139 GaussianLike 

140 A `GaussianLike` representing the distribution in the domain, with mean and 

141 covariance derived from the transformed statistics. 

142 

143 Raises 

144 ------ 

145 RuntimeError 

146 If the covariance matrix cannot be made positive definite. 

147 

148 

149 Notes 

150 ----- 

151 This method assumes that the transform is either an exactly linear or a 

152 well-defined approximately linear transformation between the codomain and the 

153 domain such that the transform can be expressed as follows: 

154 

155 .. math:: 

156 \mathcal{N}(\mu_y, \Sigma_y) 

157 \rightarrow 

158 \mathcal{N}(f^{-1}(\mu_y), \; A \Sigma_y A^\top) 

159 

160 where :math:`f^{-1}` is the inverse of the transform, :math:`\mu_y` is the mean 

161 of the distribution in the codomain, :math:`\Sigma_y` is the covariance matrix 

162 of the distribution in the codomain, and :math:`A` is the linear transformation 

163 matrix (expanded basis matrix) derived from the transform. 

164 

165 """ 

166 mean = y.mean 

167 cov = y.covariance_matrix 

168 mean_orig = self.inv(mean) 

169 assert isinstance(mean, TensorLike) 

170 assert isinstance(cov, TensorLike) 

171 assert isinstance(mean_orig, TensorLike) 

172 

173 # Get the expanded basis matrix around the mean 

174 expanded_basis_matrix = self._expanded_basis_matrix(mean) 

175 

176 # Transform covariance matrix 

177 cov_orig = expanded_basis_matrix @ cov @ expanded_basis_matrix.T 

178 

179 # Ensure positive definite 

180 cov_orig = make_positive_definite(cov_orig) 

181 

182 return GaussianLike(mean_orig, cov_orig)
