26class ConditionalNeuralProcess(RegressorMixin, BaseEstimator): 

27 """ 

28 Conditional Neural Process (CNP) Regressor. 

29 

30 This model integrates a meta-learning approach into a scikit-learn regressor. The fit() method accepts standard X, y inputs. 

31 A Conditional Neural Process (CNP) operates using context points and target points in episodes. For each episode, the model 

32 samples n_episode data points, a subset of which are designated as context points. All points from the episode, including 

33 the context points, are used as target points. Each episode thus represents a single sample in a meta-learning framework. 

34 During each epoch, we sample X.shape[0] times, ensuring that each data point is part of different contexts across episodes. 

35 The model can predict with a single input X, similar to other scikit-learn models. However, since the CNP requires context 

36 points for prediction, it stores the original X, y data as attributes, which are then used as context points for predicting 

37 new X data. 

38 

39 Parameters 

40 ---------- 

41 hidden_dim : int, default=64 

42 The number of hidden units in the neural network layers. 

43 latent_dim : int, default=64 

44 The dimensionality of the latent space. 

45 hidden_layers_enc : int, default=3 

46 The number of hidden layers in the encoder. 

47 hidden_layers_dec : int, default=3 

48 The number of hidden layers in the decoder. 

49 min_context_points : int, default=3 

50 The minimum number of context points to use during training. 

51 max_context_points : int, default=10 

52 The maximum number of context points to use during training. 

53 n_episode : int, default=32 

54 The number of episodes to sample during each epoch. 

55 max_epochs : int, default=100 

56 The maximum number of epochs to train the model. 

57 lr : float, default=0.01 

58 The learning rate for the optimizer. 

59 batch_size : int, default=16 

60 The number of samples per batch. 

61 activation : callable, default=torch.nn.ReLU 

62 The activation function to use in the neural network layers. 

63 optimizer : callable, default=torch.optim.AdamW 

64 The optimizer to use for training the model. 

65 device : str, default="cpu" 

66 The device to use for training. Options are "cpu" or "cuda". 

67 random_state : int, default=None 

68 The seed used by the random number generator. 

69 

70 References 

71 ---------- 

72 [1] Garnelo, M., Rosenbaum, D., Maddison, C., Ramalho, T., Saxton, D., Shanahan, M., Teh, Y.W., Rezende, D., & Eslami, S.M.A. (2018). 

73 Conditional Neural Processes. In International Conference on Machine Learning (pp. 1704-1713). PMLR. 

74 

75 Attributes 

76 ---------- 

77 input_dim_ : int 

78 The number of features in the input data. 

79 output_dim_ : int 

80 The number of targets in the output data. 

81 model_ : skorch.NeuralNetRegressor 

82 The neural network model used for regression. 

83 X_train_ : ndarray of shape (n_samples, n_features) 

84 The training input samples. 

85 y_train_ : ndarray of shape (n_samples,) or (n_samples, n_outputs) 

86 The target values (real numbers) in the training set. 

87 

88 Methods 

89 ------- 

90 fit(X, y) 

91 Fit the model to the training data. 

92 predict(X, return_std=False) 

93 Predict using the trained model. 

94 

95 Examples 

96 -------- 

97 >>> import numpy as np 

98 >>> from autoemulate.emulators.cnp import ConditionalNeuralProcess 

99 >>> X = np.random.rand(100, 10) 

100 >>> y = np.random.rand(100, 1) 

101 >>> cnp = ConditionalNeuralProcess(hidden_dim=128, latent_dim=128, hidden_layers=4, min_context_points=3, max_context_points=10, n_episode=32, max_epochs=100, lr=0.01, batch_size=16, activation=torch.nn.ReLU, optimizer=torch.optim.AdamW, device="cpu", random_state=42) 

102 >>> cnp.fit(X, y) 

103 >>> y_pred = cnp.predict(X) 

104 >>> y_pred.shape 

105 (100, 1) 

106 """

150 def fit(self, X, y): 

151 X, y = check_X_y( 

152 X, 

153 y, 

154 multi_output=True, 

155 dtype=np.float32, 

156 copy=True, 

157 ensure_2d=True, 

158 # ensure_min_samples=self.n_episode, 

159 y_numeric=True, 

160 ) 

161 # y also needs to be float32 and 2d 

162 y = y.astype(np.float32) 

163 self.y_dim_ = y.ndim 

164 if len(y.shape) == 1: 

165 y = y.reshape(-1, 1) 

166 

167 self.input_dim_ = X.shape[1] 

168 self.output_dim_ = y.shape[1] 

169 

170 # Normalize target value 

171 # the zero handler is from sklearn 

172 if self.normalize_y: 

173 self._y_train_mean = np.mean(y, axis=0) 

174 self._y_train_std = _handle_zeros_in_scale(np.std(y, axis=0), copy=False) 

175 

176 # Remove mean and make unit variance 

177 y = (y - self._y_train_mean) / self._y_train_std 

178 

179 if self.random_state is not None: 

180 set_random_seed(self.random_state) 

181 

182 module = CNPModule if not self.attention else AttnCNPModule 

183 self.model_ = NeuralNetRegressor( 

184 module, 

185 module__input_dim=self.input_dim_, 

186 module__output_dim=self.output_dim_, 

187 module__hidden_dim=self.hidden_dim, 

188 module__latent_dim=self.latent_dim, 

189 module__hidden_layers_enc=self.hidden_layers_enc, 

190 module__hidden_layers_dec=self.hidden_layers_dec, 

191 module__activation=self.activation, 

192 dataset__min_context_points=self.min_context_points, 

193 dataset__max_context_points=self.max_context_points, 

194 dataset__n_episode=self.n_episode, 

195 max_epochs=self.max_epochs, 

196 lr=self.lr, 

197 batch_size=self.batch_size, 

198 optimizer=self.optimizer, 

199 device=self.device, 

200 dataset=CNPDataset, # special dataset to sample context and target sets 

201 criterion=CNPLoss, 

202 iterator_train__collate_fn=cnp_collate_fn, # special collate to different n in episodes 

203 iterator_valid__collate_fn=cnp_collate_fn, 

204 callbacks=[ 

205 ("early_stopping", EarlyStopping(patience=10)), 

206 ( 

207 "lr_scheduler", 

208 LRScheduler(policy="ReduceLROnPlateau", patience=5, factor=0.5), 

209 ), 

210 ("grad_norm", GradientNormClipping(gradient_clip_value=1.0)), 

211 ], 

212 # train_split=None, 

213 verbose=0, 

214 ) 

215 self.model_.fit(X, y) 

216 self.is_fitted_ = True 

217 self.X_train_ = X 

218 self.y_train_ = y 

219 self.n_features_in_ = X.shape[1] 

220 return self

222 def predict(self, X, return_std=False): 

223 check_is_fitted(self) 

224 X = check_array(X, dtype=np.float32) 

225 X_context = torch.tensor(self.X_train_, dtype=torch.float32).unsqueeze(0) 

226 y_context = torch.tensor(self.y_train_, dtype=torch.float32).unsqueeze(0) 

227 X_target = torch.tensor(X, dtype=torch.float32).unsqueeze(0) 

228 

229 with torch.no_grad(): 

230 predictions = self.model_.module_.forward(X_context, y_context, X_target) 

231 

232 # needs to be float64 to pass estimator tests 

233 # squeeze out batch dimension again so that score() etc. runs 

234 mean, logvar = predictions 

235 

236 mean = mean[-X.shape[0] :].numpy().astype(np.float64).squeeze() 

237 logvar = logvar[-X.shape[0] :].numpy().astype(np.float64).squeeze() 

238 

239 # undo normalization 

240 if self.normalize_y: 

241 mean = mean * self._y_train_std + self._y_train_mean 

242 var = np.exp(logvar) * (self._y_train_std**2) 

243 logvar = np.log(var) 

244 

245 # make sure predictions have the same shape as y 

246 if self.y_dim_ == 1: 

247 mean = mean.ravel() 

248 logvar = logvar.ravel() 

249 else: 

250 mean = mean.reshape(-1, self.output_dim_) 

251 logvar = logvar.reshape(-1, self.output_dim_) 

252 

253 if return_std: 

254 std = np.exp(0.5 * logvar) 

255 return mean, std 

256 else: 

257 return mean
